**The Code (`convert.sh`):**

```bash
#!/bin/bash
SUBJECTS=("Architecture" "CN" "DBMS" "OS")

for sub in "${SUBJECTS[@]}"; do
    echo "ğŸ“‚ Processing: $sub"
    mkdir -p "./$sub/text"
    for file in "./$sub/pdf"/*.pdf; do
        [ -e "$file" ] || continue
        name=$(basename "$file" .pdf)
        output="./$sub/text/$name.txt"
        if [ ! -f "$output" ]; then
            pdftotext -layout -enc UTF-8 "$file" "$output"
        fi
    done
done

```
